# Training Task Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy project-scoped training tasks that persist business configuration instances and start Flyte Runs through a custom `training_task` plugin.

**Architecture:** Add a generated Connect API under `flyteidl2/trainingtask`, persist training task rows in the runs database, and mount a `TrainingTaskService` from `runs/setup.go`. Starting a training task builds an inline `TaskSpec` with task type `training_task`; the executor loads a new `flyteplugins/aione/trainingtask` plugin that creates and watches a Kubernetes `Job`. The Console gets a dedicated project nav item and pages for list, create/edit, details, and actions.

**Tech Stack:** Go, ConnectRPC/protobuf, PostgreSQL/sqlx migrations, Kubernetes client-go/controller-runtime, Next.js App Router, React, TanStack Query, Vitest, Playwright CLI, Helm/k3s deployment.

package flyteidl2.trainingtask;

import "flyteidl2/common/identifier.proto";

service TrainingTaskService {
  rpc ListTrainingTasks(ListTrainingTasksRequest) returns (ListTrainingTasksResponse) { option idempotency_level = NO_SIDE_EFFECTS; }
  rpc GetTrainingTask(GetTrainingTaskRequest) returns (GetTrainingTaskResponse) { option idempotency_level = NO_SIDE_EFFECTS; }
  rpc CreateTrainingTask(CreateTrainingTaskRequest) returns (CreateTrainingTaskResponse) {}
  rpc UpdateTrainingTask(UpdateTrainingTaskRequest) returns (UpdateTrainingTaskResponse) {}
  rpc DeleteTrainingTask(DeleteTrainingTaskRequest) returns (DeleteTrainingTaskResponse) {}
  rpc StartTrainingTask(StartTrainingTaskRequest) returns (StartTrainingTaskResponse) {}
  rpc StopTrainingTask(StopTrainingTaskRequest) returns (StopTrainingTaskResponse) {}
  rpc ListResourceSpecs(ListResourceSpecsRequest) returns (ListResourceSpecsResponse) { option idempotency_level = NO_SIDE_EFFECTS; }
  rpc ListOfficialImages(ListOfficialImagesRequest) returns (ListOfficialImagesResponse) { option idempotency_level = NO_SIDE_EFFECTS; }
}

message TrainingTaskInput {
  string name = 1;
  string description = 2;
  string resource_spec_id = 3;
  string command = 4;
  uint32 max_runtime_hours = 5;
  ImageType image_type = 6;
  string official_image_id = 7;
  string image_name = 8;
  string image_uri = 9;
}

message ListTrainingTasksRequest {
  common.ListRequest request = 1;
  common.ProjectIdentifier project = 2;
}

message ListTrainingTasksResponse {
  repeated TrainingTask training_tasks = 1;
  string token = 2;
  uint32 total = 3;
}

message GetTrainingTaskRequest { TrainingTaskIdentifier id = 1; }

message UpdateTrainingTaskRequest { TrainingTaskIdentifier id = 1; TrainingTaskInput training_task = 2; }
message UpdateTrainingTaskResponse { TrainingTask training_task = 1; }
message DeleteTrainingTaskRequest { TrainingTaskIdentifier id = 1; }
message DeleteTrainingTaskResponse {}
message StartTrainingTaskRequest { TrainingTaskIdentifier id = 1; }
message StartTrainingTaskResponse { TrainingTask training_task = 1; string run_name = 2; }
message StopTrainingTaskRequest { TrainingTaskIdentifier id = 1; string reason = 2; }
message StopTrainingTaskResponse { TrainingTask training_task = 1; }
message ListResourceSpecsRequest {}
message ListResourceSpecsResponse { repeated ResourceSpec resource_specs = 1; }
message ListOfficialImagesRequest {}
message ListOfficialImagesResponse { repeated OfficialImage official_images = 1; }
